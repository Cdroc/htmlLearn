HTML -- Hypertext Markup Language -- 超文本标记语言
                                  -- 超文本（含有指向其它文本文件链接的文本）
     -- 不是一种“编程语言”，而是一种“标记语言”
     -- 标记语言是一套“标记标签”
     -- HTML使用标记标签来描述网页
     -- 注释 -- <!-- This is a comment -->
     -- 在显示页面时，浏览器会移除源代码中多余的空格和空行，
        所有连续的空格或空行都会被算作一个空格，
        HTML代码中的所有连续的空行（换行）也被显示为一个空格
     -- &nbsp; -- 空格
     -- &lt;   -- <
     -- &gt;   -- >
     -- &#169; -- ©
-----------------------------------------------------------------------------
HTML标签 -- 由尖括号包围的关键词，比如<html>

HTML文档=网页 -- 包含HTML标签和纯文本

HTML元素 -- 指从开始标签到结束标签的所有代码
         -- 开始标签（开放标签）、结束标签（闭合标签）
         -- 元素内容 -- 开始标签与结束标签之间的内容
         -- 空元素   -- 没有内容的HTML元素
                     -- 在开始标签中进行关闭（以开始标签的结束而结束）

<body>元素 -- 定义了HTML文档的主体
<html>元素 -- 定义了整个HTML文档
<br />     -- 定义换行 -- 空行

HTML <div>和<span> -- 大多数HTML元素被定义为块级元素或内联元素
                   -- 块级元素 -- block level element
                               -- 在浏览器显示时，通常会以新行来开始（和结束）
                               -- <h1>, <p>, <ul>, <table>
                               -- <div>元素  -- 文档中的分区或节
                                             -- 可用于对大的内容块设置样式属性
                                             -- 另一个常见的用途是文档布局
                   -- 内联元素 -- inline element
                               -- 在显示时通常不会以新行开始
                               -- <b>, <td>, <a>, <img />, <input />
                               -- <span>元素 -- 用来组合文档中的行内元素
                                             -- 可用于为部分文本设置样式属性
                                             -- 可用作文本的容器
-----------------------------------------------------------------------------

                                                -- password(密码输入框)
                                                -- radio(单选按钮)，name属性值要一样才能从中选取其一
                                                -- checkbox(复选框)
                                                -- submit(确认按钮)
                                                -- reset(重置按钮)
                                                -- hidden(隐藏域)
                                                -- button(按钮)
                                 -- select(下拉列表) 
                                    -- optgroup(选项组) 
                                       -- option(选项) -- selected属性，selected="selected"(选中项)
                                 -- textarea(多行文本输入控件) -- rows属性
                                                               -- cols属性
                                 -- label(一个控制的标签)      -- for属性，值与相关元素的id值相同
                                 -- isindex(已废弃，由<input>代替)

HTML框架 -- frameset(框架结构标签) -- cols="25%,59%,25%" cols="120,*"
                                   -- rows="25%,59%,25%"
                                   -- frame(框架标签) -- src属性
                                                      -- noresize="noresize",
                                                         禁止拖动边框来改变大小
                                   -- noframes -- 为不支持框架的浏览器添加<noframes>标签
                                   -- body标签不能与frameset标签同时使用，
                                      但body标签可以放在noframes标签中
         -- iframe -- 内联的子窗口(框架) 
                   -- 一些老的浏览器不支持iframe,如果得不到支持，iframe是不可见的
                   -- <body><iframe src="url"></iframe></body>
                   -- frameborder="0",可以移除边框
                   -- iframe可用作链接a的目标，链接的target属性必须引用iframe的name属性
