HTML -- Hypertext Markup Language -- 超文本标记语言
                                  -- 超文本（含有指向其它文本文件链接的文本）
     -- 不是一种“编程语言”，而是一种“标记语言”
     -- 标记语言是一套“标记标签”
     -- HTML使用标记标签来描述网页

HTML标签 -- 由尖括号包围的关键词，比如<html>

HTML文档=网页 -- 包含HTML标签和纯文本

HTML元素 -- 指从开始标签到结束标签的所有代码
         -- 开始标签（开放标签）、结束标签（闭合标签）
         -- 元素内容 -- 开始标签与结束标签之间的内容
         -- 空元素   -- 没有内容的HTML元素
                     -- 在开始标签中进行关闭（以开始标签的结束而结束）

<body>元素 -- 定义了HTML文档的主体
<html>元素 -- 定义了整个HTML文档
<br />     -- 定义换行

HTML标题 -- h1 h2 h3 h4 h5 h6
HTML段落 -- p
HTML链接 -- a       -- href属性指定链接地址
                    -- href(hypertext reference) -- 超文本引用（超链接）
HTML图像 -- <img /> -- src属性指定图片地址
HTML表格 -- table

属性 -- align   -- 对齐方式 -- center
     -- bgcolor -- 背景颜色
     -- border  -- 边框
     -- class   -- 元素类名
     -- id      -- 元素唯一ID
     -- style   -- 行内样式
     -- title   -- 额外信息（可在工具提示中显示）
